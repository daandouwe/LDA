def load_proportions(path):
    proportions = []
    with open(path, 'r') as f:
        tuples = []
        for line in f:
            l = line.split()
            if not l:
                proportions.append(tuples)
                tuples=[]
            else:
                tuples.append((int(l[0]), float(l[1])))
    return [prop for prop in proportions if prop]

def patches(k, n_patches=-1):
    theta = load_proportions('theta.txt')
    for i, (arg, val) in enumerate(theta[k][:n_patches]):
        phi = np.load('phi-{}.npy'.format(arg))
        yield val, phi
        
def reconstruct(k):
    theta = load_proportions('theta.txt')
    result = np.zeros((28,28))
    for val, phi in patches(k):
        result += val * phi
    return result

def reconstruct_3d(k):
    theta = load_proportions('theta.txt')
    result = np.zeros((28,28,3))
    for i, (val, phi) in enumerate(patches(k, n_patches=3)):
        result[:,:,i] = val * phi
    result /= result.max() # scale for visibility in plotting
    return result

def draw_patches(k, n_patches):
    n = int(np.sqrt(n_patches))
    assert n**2 == n_patches, 'n_patches not a square number: {}'.format(n_patches)
        
    fig, axarr = plt.subplots(n, n, sharex=True, sharey=True, figsize=(2*n,2*n))
    for i, (val, phi) in enumerate(patches(k, n_patches)):
        k, i = divmod(i, n)
        axarr[k, i].imshow(phi, cmap='Greys')
        axarr[k, i].set_title('{:.3f}'.format(val))
    plt.show()

def draw_all_patches():
    n = 5
    fig, axarr = plt.subplots(n, n, sharex=True, sharey=True, figsize=(2*n,2*n))
    for i in range(int(n**2)):
        phi = np.load('phi-{}.npy'.format(i))
        k, i = divmod(i, n)
        axarr[k, i].imshow(phi, cmap='Greys')
    plt.show()

from matplotlib.colors import to_rgb

color_names = ['tab:blue', 'tab:orange', 'tab:green', 'tab:red', 'tab:purple', 
               'tab:brown', 'tab:pink', 'tab:gray', 'tab:olive', 'tab:cyan']

colors = [to_rgb(c) for c in color_names]
        
def reconstruct_color(k, colors, n_patches=3):
    theta = load_proportions('theta.txt')
    result = np.zeros((28,28,3))
    for i, (val, phi) in enumerate(patches(k, n_patches)):
        color = colors[i]
        phi = np.dstack([c*(phi) for c in color])
        result += val * phi
    result /= result.max() # scale for visibility in plotting
    return result

result = reconstruct_color(0, colors, n_patches=5)
draw_color(result)
draw_color(1-result)
